clock-timer.js

Display a really basic animated clock with JavaScript.

Github | Download

Basic

window.clock = new ClockTimer();

Countdown

	var options = {
		endDate: "12/31/2018 24:00:00" // UTC
	};

	window.clock = new ClockTimer( options );

Between Two Dates

var options = {
	startDate: "10/31/2018 24:00:00",
	endDate: "12/24/2018 24:00:00"
};

window.clock = new ClockTimer( options );

Customize

var options = {
		endDate: "12/24/2018 24:00:00",
		lineWidth: 2,
		radius: 120,
		secondsStrokeStyle: "#C7181D",
		minutesStrokeStyle: "#FCB937",
		hoursStrokeStyle: "#A1B836",
		daysStrokeStyle: "#371979"
};

window.clock = new ClockTimer( options );

Count only

	var options = {
		endDate: "12/24/2018 24:00:00",
		secondsCanvas: false,
		minutesCanvas: false,
		hoursCanvas: false,
		daysCanvas: false,
	};

	window.clock = new ClockTimer( options );
